 (lambda (row)
   (cl-destructuring-bind (соч фун) row
     (global-set-key (kbd соч) (intern фун))))
 keys-table)

(-map
 (lambda (row)
   (cl-destructuring-bind (соч фун) row
     (global-set-key (kbd соч) (intern фун))))
 exwm-keys-table)

    (progn
	    (require 'exwm)
	    (-map
	     (lambda (row)
	       (cl-destructuring-bind (соч фун) row
	         (exwm-input-set-key (kbd соч) (intern фун))))
	     exwm-keys-table)))
